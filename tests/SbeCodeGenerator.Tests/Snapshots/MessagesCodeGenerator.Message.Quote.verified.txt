using System.Runtime.InteropServices;
namespace TestNamespace;
/// <summary>
/// Quote message
/// (MessageDefinition)
/// </summary>
[StructLayout(LayoutKind.Explicit, Pack = 1)]
public partial struct QuoteData
{
	/// <summary>
	/// Message Id
	/// (ConstantMessageFieldDefinition)
	/// </summary>
	public const int MESSAGE_ID = 2;
	/// <summary>
	/// Message Size
	/// (ConstantMessageFieldDefinition)
	/// </summary>
	public const int MESSAGE_SIZE = 32;
	/// <summary>
	/// 
	/// (MessageFieldDefinition)
	/// </summary>
	[FieldOffset(0)]
	public long BidPrice;
	/// <summary>
	/// 
	/// (MessageFieldDefinition)
	/// </summary>
	[FieldOffset(8)]
	public long BidQuantity;
	/// <summary>
	/// 
	/// (MessageFieldDefinition)
	/// </summary>
	[FieldOffset(16)]
	public long AskPrice;
	/// <summary>
	/// 
	/// (MessageFieldDefinition)
	/// </summary>
	[FieldOffset(24)]
	public long AskQuantity;
	public static bool TryParse(ReadOnlySpan<byte> buffer, out QuoteData message, out ReadOnlySpan<byte> variableData)
	{
		if (buffer.Length < MESSAGE_SIZE)
		{
			message = default;
			variableData = default;
			return false;
		}
		message = MemoryMarshal.AsRef<QuoteData>(buffer);
		variableData = buffer.Slice(MESSAGE_SIZE);
		return true;
	}
}
